00160                 {

00164                                 success = false;

00167 
00168                 //currently, i demand that all input slots of a service are either set to sth or linked before starting
00169                 for (InputMap::const_iterator it = m_InputSlots.begin(); it != m_InputSlots.end(); ++it)
00170                 {
00171                         if (!it->second->isInitialized() && !it->second->isLinked())
00172                         {
00173                                 msg << "input slot " << it->second->getName() << " of service " << name() << " has been neither set nor linked\n";
00174                                 success = false;
00175                         }
00176                 }
00177 
00178                 if (!success)
00179                 {
00180                         throw StartException(msg.str());
